 478    {
 479        /// <summary>The scheduler on which to run the action.</summary>

 482        /// <summary>Initializes the TaskSchedulerAwaitTaskContinuation.</summary>
 483        /// <param name="scheduler">The task scheduler with which to invoke the action.  Must not be null.</param>
 484        /// <param name="action">The action to invoke. Must not be null.</param>

 486        /// <param name="stackMark">The captured stack mark.</param>
 487        [SecurityCritical]
 488        internal TaskSchedulerAwaitTaskContinuation(
 489            TaskScheduler scheduler, Action action, bool flowExecutionContext, ref StackCrawlMark stackMark) :
 490            base(action, flowExecutionContext, ref stackMark)
 491        {
 492            Contract.Assert(scheduler != null);
 493            m_scheduler = scheduler;
 494        }
 495
 496        /// <summary>Inlines or schedules the continuation.</summary>
 497        /// <param name="ignored">The antecedent task, which is ignored.</param>
 498        /// <param name="canInlineContinuationTask">true if inlining is permitted; otherwise, false.</param>
 499        internal sealed override void Run(Task ignored, bool canInlineContinuationTask)
 500        {
 501            // If we're targeting the default scheduler, we can use the faster path provided by the base class.
 502            if (m_scheduler == TaskScheduler.Default)
 503            {
 504                base.Run(ignored, canInlineContinuationTask);
 505            }
 506            else
 507            {
 508                // We permit inlining if the caller allows us to, and
 509                // either we're on a thread pool thread (in which case we're fine running arbitrary code)
 510                // or we're already on the target scheduler (in which case we'll just ask the scheduler
 511                // whether it's ok to run here).  We include the IsThreadPoolThread check here, whereas
 512                // we don't in AwaitTaskContinuation.Run, since here it expands what's allowed as opposed
 513                // to in AwaitTaskContinuation.Run where it restricts what's allowed.
 514                bool inlineIfPossible = canInlineContinuationTask &&
 515                    (TaskScheduler.InternalCurrent == m_scheduler || Thread.CurrentThread.IsThreadPoolThread);
 516
 517                // Create the continuation task task. If we're allowed to inline, try to do so.
 518                // The target scheduler may still deny us from executing on this thread, in which case this'll be queued
 519                var task = CreateTask(state => {
 520                    try { ((Action)state)(); }
 521                    catch (Exception exc) { ThrowAsyncIfNecessary(exc); }
 522                }, m_action, m_scheduler);
 523
 524                if (inlineIfPossible)
 525                {
 526                    InlineIfPossibleOrElseQueue(task, needsProtection: false);
 527                }
 528                else
 529                {
 530                    // We need to run asynchronously, so just schedule the task.
 531                    try { task.ScheduleAndStart(needsProtection: false); }
 532                    catch (TaskSchedulerException) { } // No further action is necessary, as ScheduleAndStart already tr
 533                }
 534            }
 535        }
 536    }

 626        /// <summary>
 627        /// Gets whether the current thread is an appropriate location to inline a continuation's execution.
 628        /// </summary>
 629        /// <remarks>
 630        /// Returns whether SynchronizationContext is null and we're in the default scheduler.
 631        /// If the await had a SynchronizationContext/TaskScheduler where it began and the
 632        /// default/ConfigureAwait(true) was used, then we won't be on this path.  If, however,
 633        /// ConfigureAwait(false) was used, or the SynchronizationContext and TaskScheduler were
 634        /// naturally null/Default, then we might end up here.  If we do, we need to make sure
 635        /// that we don't execute continuations in a place that isn't set up to handle them, e.g.
 636        /// running arbitrary amounts of code on the UI thread.  It would be "correct", but very
 637        /// expensive, to always run the continuations asynchronously, incurring lots of context
 638        /// switches and allocations and locks and the like.  As such, we employ the heuristic
 639        /// that if the current thread has a non-null SynchronizationContext or a non-default
 640        /// scheduler, then we better not run arbitrary continuations here.
 641        /// </remarks>
 642        internal static bool IsValidLocationForInlining
 643        {
 644            get
 645            {
 646                // If there's a SynchronizationContext, we'll be conservative and say
 647                // this is a bad location to inline.
 648                var ctx = SynchronizationContext.CurrentNoFlow;
 649                if (ctx != null && ctx.GetType() != typeof(SynchronizationContext)) return false;
 650
 651                // Similarly, if there's a non-default TaskScheduler, we'll be conservative
 652                // and say this is a bad location to inline.
 653                var sched = TaskScheduler.InternalCurrent;
 654                return sched == null || sched == TaskScheduler.Default;
 655            }
 656        }

 732            ContextCallback callback = s_invokeActionCallback;

 737        /// <summary>Runs the callback synchronously with the provided state.</summary>
 738        /// <param name="callback">The callback to run.</param>
 739        /// <param name="state">The state to pass to the callback.</param>
 740        /// <param name="currentTask">A reference to Task.t_currentTask.</param>
 741        [SecurityCritical]
 742        protected void RunCallback(ContextCallback callback, object state, ref Task currentTask)
 743        {
 744            Contract.Requires(callback != null);
 745            Contract.Assert(currentTask == Task.t_currentTask);
 746
 747            // Pretend there's no current task, so that no task is seen as a parent
 748            // and TaskScheduler.Current does not reflect false information
 749            var prevCurrentTask = currentTask;
 750            try
 751            {
 752                if (prevCurrentTask != null) currentTask = null;
 753
 754                // If there's no captured context, just run the callback directly.
 755                if (m_capturedContext == null) callback(state);
 756                // Otherwise, use the captured context to do so.
 757                else ExecutionContext.Run(m_capturedContext, callback, state, true);
 758            }
 759            catch (Exception exc) // we explicitly do not request handling of dangerous exceptions like AVs
 760            {
 761                ThrowAsyncIfNecessary(exc);
 762            }
 763            finally
 764            {
 765                // Restore the current task information
 766                if (prevCurrentTask != null) currentTask = prevCurrentTask;
 767
 768                // Clean up after the execution context, which is only usable once.
 769                if (m_capturedContext != null) m_capturedContext.Dispose();
 770            }
 771        }

 773        /// <summary>Invokes or schedules the action to be executed.</summary>
 774        /// <param name="action">The action to invoke or queue.</param>
 775        /// <param name="allowInlining">
 776        /// true to allow inlining, or false to force the action to run asynchronously.
 777        /// </param>
 778        /// <param name="currentTask">
 779        /// A reference to the t_currentTask thread static value.
 780        /// This is passed by-ref rather than accessed in the method in order to avoid
 781        /// unnecessary thread-static writes.
 782        /// </param>
 783        /// <remarks>
 784        /// No ExecutionContext work is performed used.  This method is only used in the
 785        /// case where a raw Action continuation delegate was stored into the Task, which
 786        /// only happens in Task.SetContinuationForAwait if execution context flow was disabled
 787        /// via using TaskAwaiter.UnsafeOnCompleted or a similar path.
 788        /// </remarks>
 789        [SecurityCritical]
 790        internal static void RunOrScheduleAction(Action action, bool allowInlining, ref Task currentTask)
 791        {
 792            Contract.Assert(currentTask == Task.t_currentTask);
 793
 794            // If we're not allowed to run here, schedule the action
 795            if (!allowInlining || !IsValidLocationForInlining)
 796            {
 797                UnsafeScheduleAction(action, currentTask);
 798                return;
 799            }
 800
 801            // Otherwise, run it, making sure that t_currentTask is null'd out appropriately during the execution
 802            Task prevCurrentTask = currentTask;
 803            try
 804            {
 805                if (prevCurrentTask != null) currentTask = null;
 806                action();
 807            }
 808            catch (Exception exception)
 809            {
 810                ThrowAsyncIfNecessary(exception);
 811            }
 812            finally
 813            {
 814                if (prevCurrentTask != null) currentTask = prevCurrentTask;
 815            }
